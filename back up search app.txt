function get_build($con,$tp,$dt){
	$sel_1=mysqli_query($con,"select build from current_tp where name='$tp' and date='$dt';");

		$b=mysqli_fetch_array($sel_1);
		$build=$b[0];
		if($build==NULL){
			$build="Not Available";
		}
	}

		$b=mysqli_fetch_array($sel_2);
		$build=$b[0];
		if($build==NULL){
			$build="Not Available";
		}
	}
	else{
		$build="Not Available";
	}
	return $build;
